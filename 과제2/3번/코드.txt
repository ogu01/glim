#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> dots) {
    int answer = 0;
    
    int sumy = 0;
    int sumx = 0;
    for(int i = 1; i < 4; i++){
        if(dots[0][0] == dots[i][0]){
            if(dots[0][1]>dots[i][1]){
               sumy = dots[0][1]-dots[i][1];
            }
            else{
                 sumy = dots[i][1] - dots[0][1];
            }
        }
        
        if(dots[0][1] == dots[i][1]){
            if(dots[0][0]>dots[i][0]){
               sumx = dots[0][0]-dots[i][0];
            }
            else{
                 sumx = dots[i][0] - dots[0][0];
            }
        }
        
    }
        answer = sumx * sumy;
        
    return answer;
}
