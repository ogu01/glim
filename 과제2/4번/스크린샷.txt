#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> arr, vector<bool> flag) {
    vector<int> answer;
    for(int i = 0; i < arr.size(); i++){
        if(flag[i]){
           for(int a = 0; a < arr[i]*2; a++){
               answer.push_back(arr[i]);
           }
        }
        else{
            for(int b = 0; b < arr[i]; b++){
                answer.pop_back();
            }
        }
    }
    
    
    return answer;
}
